# TypeScript 装饰器详解

## 一、装饰器的定义

**装饰器（Decorator）**是一种特殊的声明，可以附加到类、方法、属性、参数和访问符上，用于修改或扩展其行为。

装饰器的本质是一个**函数**，接受目标对象作为参数，返回修改后的对象或元数据。

```typescript
// 装饰器的基本语法
@decoratorName
class MyClass { }
```

## 二、如何理解装饰器

### 2.1 核心概念

装饰器模式是一种设计模式，允许在不修改原始代码的情况下，动态地为对象添加功能。在 TypeScript 中，装饰器提供了一种声明式的语法来实现这一模式。

**类比理解**：
- 就像给手机贴膜、戴壳，不改变手机本身，但增加了保护功能
- 就像给函数包装一层逻辑（高阶函数），在执行前后添加额外行为

### 2.2 执行时机

装饰器在**编译时**（而非运行时）执行，这意味着：
- 装饰器在代码定义阶段就会应用
- 可以在类实例化前修改类的结构和行为

## 三、装饰器的使用

### 3.1 启用装饰器

在 `tsconfig.json` 中启用：

```json
{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}
```

### 3.2 类装饰器（Class Decorator）

```typescript
// 类装饰器定义
function ClassDecorator(constructor: Function) {
  console.log('类装饰器被调用');
  // 可以修改类原型或类本身
  constructor.prototype.newMethod = function() {
    console.log('添加的新方法');
  }
}

@ClassDecorator
class MyClass {
  constructor() {
    console.log('类被实例化');
  }
}

const obj = new MyClass();
obj.newMethod(); // 添加的新方法
```

**类装饰器工厂**（带参数的装饰器）：

```typescript
function AddMetadata(metadata: string) {
  return function(constructor: Function) {
    constructor.prototype.metadata = metadata;
  }
}

@AddMetadata('这是元数据')
class UserService { }

console.log(new UserService().metadata); // 这是元数据
```

### 3.3 方法装饰器（Method Decorator）

```typescript
/**
 * 方法装饰器参数：
 * target: 对于静态方法是类的构造函数，对于实例方法是类的原型
 * propertyKey: 方法名称
 * descriptor: 属性描述符
 */
function Log(
  target: any,
  propertyKey: string,
  descriptor: PropertyDescriptor
) {
  const originalMethod = descriptor.value;

  descriptor.value = function(...args: any[]) {
    console.log(`调用方法 ${propertyKey}，参数：`, args);
    const result = originalMethod.apply(this, args);
    console.log(`方法 ${propertyKey} 返回：`, result);
    return result;
  };
}

class Calculator {
  @Log
  add(a: number, b: number): number {
    return a + b;
  }
}

const calc = new Calculator();
calc.add(1, 2);
// 输出：
// 调用方法 add，参数：[1, 2]
// 方法 add 返回：3
```

### 3.4 属性装饰器（Property Decorator）

```typescript
/**
 * 属性装饰器参数：
 * target: 对于静态属性是类的构造函数，对于实例属性是类的原型
 * propertyKey: 属性名称
 */
function ReadOnly(target: any, propertyKey: string) {
  Object.defineProperty(target, propertyKey, {
    writable: false
  });
}

function DefaultValue(value: any) {
  return function(target: any, propertyKey: string) {
    let _value = value;
    
    const getter = () => _value;
    const setter = (newValue: any) => {
      console.warn(`属性 ${propertyKey} 是只读的`);
    };
    
    Object.defineProperty(target, propertyKey, {
      get: getter,
      set: setter,
      enumerable: true,
      configurable: true
    });
  };
}

class User {
  @ReadOnly
  id: number = 1;

  @DefaultValue('匿名用户')
  name: string;
}

const user = new User();
user.id = 2; // TypeScript 中可能报错，运行时不生效
console.log(user.id); // 1
```

### 3.5 参数装饰器（Parameter Decorator）

```typescript
/**
 * 参数装饰器参数：
 * target: 构造函数或原型
 * propertyKey: 方法名称（undefined 对于构造函数参数）
 * parameterIndex: 参数在参数列表中的索引
 */
function Required(target: any, propertyKey: string, parameterIndex: number) {
  console.log(`参数装饰器：方法 ${propertyKey} 的第 ${parameterIndex} 个参数是必需的`);
}

function Validate(target: any, propertyKey: string, parameterIndex: number) {
  const validator = (value: any) => {
    if (value === undefined || value === null) {
      throw new Error(`参数 ${parameterIndex} 不能为空`);
    }
  };

  // 存储验证器（需要配合方法装饰器使用）
  if (!target._validators) {
    target._validators = {};
  }
  if (!target._validators[propertyKey]) {
    target._validators[propertyKey] = [];
  }
  target._validators[propertyKey].push({
    index: parameterIndex,
    validator
  });
}

class UserService {
  createUser(@Required name: string, @Validate age: number) {
    console.log(`创建用户：${name}, ${age}岁`);
  }
}
```

### 3.6 访问符装饰器（Accessor Decorator）

```typescript
function Enumerable(enumerable: boolean) {
  return function(
    target: any,
    propertyKey: string,
    descriptor: PropertyDescriptor
  ) {
    descriptor.enumerable = enumerable;
  };
}

class Person {
  private _name: string;

  constructor(name: string) {
    this._name = name;
  }

  @Enumerable(false)
  get name() {
    return this._name;
  }

  set name(value: string) {
    this._name = value;
  }
}
```

## 四、装饰器的作用

### 4.1 AOP（面向切面编程）

实现横切关注点的分离，如日志、权限、缓存等：

```typescript
// 日志装饰器
function LogMethod(target: any, key: string, descriptor: PropertyDescriptor) {
  const original = descriptor.value;
  descriptor.value = function(...args: any[]) {
    const start = Date.now();
    const result = original.apply(this, args);
    const end = Date.now();
    console.log(`${key} 执行时间：${end - start}ms`);
    return result;
  };
}

// 性能监控装饰器
function PerformanceMonitor(threshold: number) {
  return function(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function(...args: any[]) {
      const start = performance.now();
      const result = original.apply(this, args);
      const end = performance.now();
      if (end - start > threshold) {
        console.warn(`${key} 执行缓慢：${(end - start).toFixed(2)}ms`);
      }
      return result;
    };
  };
}
```

### 4.2 元数据注入

为类或属性添加元数据，常用于框架开发：

```typescript
function Component(selector: string) {
  return function(constructor: Function) {
    constructor.prototype.selector = selector;
  };
}

@Component('#app')
class AppComponent {
  // ...
}
```

### 4.3 权限控制

```typescript
function RequireAuth(role: string) {
  return function(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function(...args: any[]) {
      const currentUser = getCurrentUser(); // 假设的函数
      if (!currentUser || currentUser.role !== role) {
        throw new Error('权限不足');
      }
      return original.apply(this, args);
    };
  };
}

class AdminService {
  @RequireAuth('admin')
  deleteUser(userId: string) {
    // 删除用户逻辑
  }
}
```

### 4.4 缓存

```typescript
function Memoize(target: any, key: string, descriptor: PropertyDescriptor) {
  const original = descriptor.value;
  const cache = new Map();
  
  descriptor.value = function(...args: any[]) {
    const cacheKey = JSON.stringify(args);
    if (cache.has(cacheKey)) {
      return cache.get(cacheKey);
    }
    const result = original.apply(this, args);
    cache.set(cacheKey, result);
    return result;
  };
}

class MathService {
  @Memoize
  fibonacci(n: number): number {
    if (n <= 1) return n;
    return this.fibonacci(n - 1) + this.fibonacci(n - 2);
  }
}
```

### 4.5 防抖和节流

```typescript
function Debounce(delay: number) {
  return function(target: any, key: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    let timer: NodeJS.Timeout | null = null;
    
    descriptor.value = function(...args: any[]) {
      if (timer) clearTimeout(timer);
      timer = setTimeout(() => {
        original.apply(this, args);
      }, delay);
    };
  };
}

class SearchComponent {
  @Debounce(300)
  onSearchInput(query: string) {
    // 执行搜索
  }
}
```

## 五、装饰器的影响范围

### 5.1 执行顺序

多个装饰器应用于同一声明时，执行顺序如下：

```typescript
function First() {
  console.log('First: 工厂函数');
  return function(target: any, propertyKey: string, descriptor?: PropertyDescriptor) {
    console.log('First: 装饰器函数');
  };
}

function Second() {
  console.log('Second: 工厂函数');
  return function(target: any, propertyKey: string, descriptor?: PropertyDescriptor) {
    console.log('Second: 装饰器函数');
  };
}

class Example {
  @First()
  @Second()
  method() {}
}

// 输出：
// First: 工厂函数
// Second: 工厂函数
// Second: 装饰器函数
// First: 装饰器函数
```

**总结：**
- 工厂函数：从上到下
- 装饰器函数：从下到上

### 5.2 不同类型装饰器的执行顺序

```typescript
// 1. 类装饰器最后执行
// 2. 成员装饰器先执行
// 执行顺序：参数装饰器 → 方法/属性装饰器 → 类装饰器
```

### 5.3 作用域限制

```typescript
// 实例成员装饰器只影响实例
class Example {
  @LogInstance
  instanceMethod() {} // 只影响实例方法

  @LogStatic
  static staticMethod() {} // 只影响静态方法
}
```

## 六、注意事项

### 6.1 TypeScript 版本

- TypeScript 5.0+ 使用新的装饰器标准（ECMAScript Stage 3）
- 旧版本使用实验性装饰器
- 两种标准不兼容，需要注意版本选择

```json
// tsconfig.json
{
  "compilerOptions": {
    "target": "ES2022",
    "experimentalDecorators": false, // 新标准不需要
    "decorators": true // 新标准
  }
}
```

### 6.2 类型安全

装饰器可能破坏类型推断：

```typescript
function AddMethod<T extends { new (...args: any[]): {} }>(
  constructor: T
) {
  return class extends constructor {
    newMethod() {
      console.log('新方法');
    }
  };
}

@AddMethod
class MyClass {
  existingMethod() {}
}

const obj = new MyClass();
obj.existingMethod(); // OK
obj.newMethod(); // TypeScript 可能不知道这个方法
```

**解决方案**：使用类型声明合并

```typescript
declare interface MyClass {
  newMethod(): void;
}
```

### 6.3 this 指向问题

在方法装饰器中要注意 this 的绑定：

```typescript
function Log(target: any, key: string, descriptor: PropertyDescriptor) {
  const original = descriptor.value;
  descriptor.value = function(...args: any[]) {
    // 这里的 this 指向实例
    console.log('this:', this);
    return original.apply(this, args);
  };
}
```

### 6.4 性能考虑

- 装饰器在编译时执行，但添加的包装函数在每次调用时都会执行
- 过度使用装饰器可能影响性能
- 避免在装饰器中执行重操作

### 6.5 调试困难

- 装饰器改变了代码结构，可能增加调试难度
- 堆栈追踪可能变得复杂
- 建议在装饰器中添加有意义的日志

### 6.6 兼容性

- 装饰器目前是 ECMAScript 提案阶段
- 需要编译器支持（如 Babel、TypeScript）
- 运行时可能需要 polyfill

### 6.7 最佳实践

1. **单一职责**：每个装饰器只做一件事
2. **可组合性**：装饰器应该可以独立使用和组合
3. **文档化**：为装饰器添加清晰的文档
4. **类型安全**：尽量提供完整的类型定义
5. **性能优化**：避免在热路径上使用复杂装饰器

```typescript
// 好的装饰器设计
export function Cached(ttl?: number): MethodDecorator {
  return (target, key, descriptor) => {
    const original = descriptor.value!;
    const cache = new Map<string, { value: any; expires: number }>();

    descriptor.value = function(this: any, ...args: any[]) {
      const cacheKey = JSON.stringify(args);
      const cached = cache.get(cacheKey);

      if (cached && (ttl === undefined || Date.now() < cached.expires)) {
        return cached.value;
      }

      const result = original.apply(this, args);
      cache.set(cacheKey, {
        value: result,
        expires: ttl ? Date.now() + ttl : Infinity
      });

      return result;
    };

    // 保留原始方法的元数据
    Object.defineProperty(descriptor.value, 'name', {
      value: original.name
    });

    return descriptor;
  };
}
```

## 七、实际应用示例

### 7.1 简单的依赖注入容器

```typescript
const Container = new Map();

function Injectable(token: string) {
  return function(constructor: Function) {
    Container.set(token, constructor);
  };
}

function Inject(token: string) {
  return function(target: any, propertyKey: string) {
    Object.defineProperty(target, propertyKey, {
      get() {
        const Service = Container.get(token);
        if (!Service) {
          throw new Error(`Service ${token} not found`);
        }
        return new Service();
      }
    });
  };
}

@Injectable('userService')
class UserService {
  getUsers() {
    return ['Alice', 'Bob'];
  }
}

class UserController {
  @Inject('userService')
  userService!: UserService;

  listUsers() {
    return this.userService.getUsers();
  }
}
```

### 7.2 路由装饰器（类似 Express/Angular）

```typescript
const routes: { path: string; method: string; handler: Function }[] = [];

function Get(path: string) {
  return function(target: any, key: string, descriptor: PropertyDescriptor) {
    routes.push({ path, method: 'GET', handler: descriptor.value });
  };
}

function Post(path: string) {
  return function(target: any, key: string, descriptor: PropertyDescriptor) {
    routes.push({ path, method: 'POST', handler: descriptor.value });
  };
}

@Controller('/users')
class UserController {
  @Get('/')
  findAll() {
    return '获取所有用户';
  }

  @Post('/')
  create() {
    return '创建用户';
  }
}

// routes 现在包含所有路由信息
```

## 八、总结

### 优点
✅ 声明式编程，代码更简洁
✅ 关注点分离，逻辑更清晰
✅ 可复用性强
✅ 适合框架开发
✅ 易于组合和扩展

### 缺点
❌ 学习曲线较陡
❌ 调试相对困难
❌ 可能影响性能
❌ 类型安全需要额外处理
❌ 标准尚未完全稳定

### 何时使用
- 开发框架或库
- 需要统一处理横切关注点
- 需要元数据驱动
- 简化重复性代码

### 何时不使用
- 简单的增删改查
- 性能敏感场景
- 团队不熟悉装饰器
- 过度设计
