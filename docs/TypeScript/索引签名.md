### 索引签名
* 允许我们为对象的动态属性定义类型约束
### 一。 字符串索引签名
#### 1. 基本用法
  ```typescript
  interface UserMap {
  [userId: string]: {
    name: string;
    age: number;
  };
  }

  const users: UserMap = {
    "user1": { name: "Alice", age: 25 },
    "user2": { name: "Bob", age: 30 },
  };

  // 可以动态添加属性
  users["user3"] = { name: "Charlie", age: 35 };

  // 访问属性
  const userName = users["user1"].name;  // "Alice"

  // 遍历
  for (const userId in users) {
    console.log(users[userId]);
  }
  ```
  #### 2. 类型约束
  ```typescript
  // 值类型可以是联合类型
interface Config {
  [key: string]: string | number | boolean;
}

const config: Config = {
  theme: "dark",
  fontSize: 14,
  debug: true,
  maxItems: 100,
};

// 更严格的类型约束
interface StrictConfig {
  [key: string]: string | number | boolean;
  name: string;  // ✅ 可以添加具体的属性
  enabled: boolean;
  // data: object;  // ❌ 必须是 string | number | boolean 之一
}
  ```
### 二。 数字索引签名
#### 1. 数组替代方案
```typescript
// 类似于数组，但更灵活
interface StringArray {
  [index: number]: string;
}

const arr: StringArray = ["a", "b", "c"];
arr[0] = "A";  // ✅
arr[100] = "Z"; // ✅ 可以任意索引

// 与数组的区别
const normalArray: string[] = ["a", "b", "c"];
normalArray[100] = "Z";  // 允许，但会创建稀疏数组
```
#### 2. 类数组对象
```typescript
interface ArgumentsLike {
  [index: number]: any;
  length: number;
  callee?: Function;
}

function example() {
  const args: ArgumentsLike = arguments;
  for (let i = 0; i < args.length; i++) {
    console.log(args[i]);
  }
}
```
### 3. 高级用法
#### 1. 与泛型结合
```typescript
// 创建通用的字典类型
interface Dictionary<T> {
  [key: string]: T;
}

// 使用示例
const stringDict: Dictionary<string> = {
  a: "hello",
  b: "world",
};

const numberDict: Dictionary<number> = {
  x: 1,
  y: 2,
};

// 嵌套字典
const nestedDict: Dictionary<Dictionary<number>> = {
  group1: { a: 1, b: 2 },
  group2: { c: 3, d: 4 },
};
```
#### 2. 映射类型中使用索引签名
```typescript
// 将所有属性值转换为 Promise
type Asyncify<T> = {
  [K in keyof T]: Promise<T[K]>;
};

// 将索引签名与其他属性结合
interface ApiResponse {
  [key: string]: any;
  status: number;
  message: string;
}

const response: ApiResponse = {
  status: 200,
  message: "OK",
  data: { id: 1, name: "Alice" },  // 动态属性
  metadata: { timestamp: "2024-01-01" },
};
```
### 4. 类型安全技巧
#### 1. 类型保护
```typescript
interface DynamicObject {
  [key: string]: string | number | boolean;
}

function getValue(obj: DynamicObject, key: string): string | number | boolean {
  return obj[key];
}

// 类型守卫
function isString(value: any): value is string {
  return typeof value === "string";
}

const value = getValue(config, "theme");
if (isString(value)) {
  console.log(value.toUpperCase());  // 类型安全
}
```
#### 2. 键值类型约束
```typescript
// 限制键的类型
type ValidKeys = "name" | "email" | "age";

interface StrictObject {
  [K in ValidKeys]: string | number;
}

// 或者使用联合类型
interface FlexibleObject {
  [key: string]: string | number;
  // 必须包含的属性
  id: string;
  createdAt: string;
}
```
#### 3. 避免类型断言
```typescript
// 不好的做法：频繁使用类型断言
const badDict: { [key: string]: any } = {};
const value1 = badDict["key"] as string;

// 好的做法：使用更精确的类型
interface TypedDict {
  [key: string]: string | undefined;
}

const goodDict: TypedDict = {};
const value2 = goodDict["key"];  // string | undefined
if (value2) {
  console.log(value2.toUpperCase());  // 类型安全
}
```
### 5. 注意事项
#### 1. 索引签名覆盖所有属性
```typescript
interface Problematic {
  [key: string]: string;
  name: string;      // ✅ 正确
  age: number;       // ❌ 错误，必须是 string 类型
}
```
#### 2. 可选属性处理
```typescript
interface WithOptional {
  [key: string]: string | undefined;
  required: string;
  optional?: string;  // ✅ 正确，因为 undefined 是允许的
}
```
#### 3. 方法定义
```typescript
interface WithMethods {
  [key: string]: Function | string;
  
  // 可以定义具体方法
  greet: (name: string) => string;
  // 也可以有属性
  name: string;
}
```
