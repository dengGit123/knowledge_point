### 1. 接口 (Interface)
* 接口定义对象的形状（结构），不包含实现
```typescript
interface Person {
  readonly id: number;     // 只读属性
  name: string;            // 必需属性
  age?: number;            // 可选属性
  [key: string]: any;      // 索引签名
  
  // 方法签名
  greet(): string;
}

// 使用接口作为类型
const person: Person = {
  id: 1,
  name: "Alice",
  greet() {
    return `Hello, I'm ${this.name}`;
  }
};

// 接口扩展（继承）
interface Employee extends Person {
  employeeId: string;
  department: string;
  
  // 可以添加新方法
  work(): void;
}

const employee: Employee = {
  id: 1,
  name: "Bob",
  employeeId: "E123",
  department: "Engineering",
  greet() {
    return `Hello, I'm ${this.name} from ${this.department}`;
  },
  work() {
    console.log("Working...");
  }
};

// 多个接口继承
interface Named {
  name: string;
}

interface Aged {
  age: number;
}

interface PersonInfo extends Named, Aged {
  email: string;
}
```
### 2. 函数类型接口
* 定义函数类型
```typescript
// 定义函数类型
interface SearchFunction {
  (source: string, keyword: string): boolean;
}

// 实现函数
const search: SearchFunction = function(src, kw) {
  return src.indexOf(kw) > -1;
};

// 箭头函数也可以
const searchArrow: SearchFunction = (src, kw) => src.includes(kw);

console.log(search("hello world", "world"));  // true
```
### 3. 可索引类型接口
* 定义索引签名
```typescript
// 数字索引签名
interface StringArray {
  [index: number]: string;
}

const arr: StringArray = ["a", "b", "c"];
console.log(arr[0]);  // "a"

// 字符串索引签名
interface NumberDictionary {
  [key: string]: number;
}

const dict: NumberDictionary = {
  length: 3,
  apples: 10,
  bananas: 5
};
```
