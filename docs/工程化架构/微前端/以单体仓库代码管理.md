# Monorepo 方式管理微前端

> 使用 Monorepo 管理微前端项目，实现代码共享、统一构建、简化协作。

## 一、方案概述

### 1.1 为什么选择 Monorepo + 微前端

```
┌─────────────────────────────────────────────────────────────────┐
│                    传统 Multi-Repo 方案                         │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  repo-a.com       repo-b.com       repo-c.com                  │
│  ┌─────────┐     ┌─────────┐     ┌─────────┐                   │
│  │ App A   │     │ App B   │     │ App C   │                   │
│  └─────────┘     └─────────┘     └─────────┘                   │
│       │               │               │                        │
│       └───────────────┼───────────────┘                        │
│                       ↓                                        │
│              需要发布 npm 包共享代码                            │
│              各自独立构建，重复安装依赖                         │
│              跨项目重构需要多个仓库同步                         │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────────────┐
│                    Monorepo 微前端方案                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│                    micro-frontend/                             │
│                    ├── main-app/                               │
│                    │   └── qiankun 主应用                       │
│                    ├── apps/                                   │
│                    │   ├── app-vue/                            │
│                    │   ├── app-react/                          │
│                    │   └── app-admin/                          │
│                    └── packages/                               │
│                        ├── shared/     ← 共享工具               │
│                        ├── types/      ← 共享类型               │
│                        ├── components/ ← 共享组件               │
│                        └── config/     ← 共享配置               │
│                                                                 │
│              ✅ 代码直接引用，实时更新                          │
│              ✅ 统一依赖管理，去重安装                          │
│              ✅ 一次 PR 完成跨项目重构                          │
│              ✅ 共享构建配置和 CI 流程                          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 技术选型

| 工具 | 推荐指数 | 说明 |
|------|----------|------|
| **pnpm** | ⭐⭐⭐⭐⭐ | 节省空间、符号链接、速度快 |
| **TurboRepo** | ⭐⭐⭐⭐ | 增量构建、缓存优化 |
| **Nx** | ⭐⭐⭐ | 功能全面、学习成本高 |
| **Lerna** | ⭐⭐ | 老牌工具、配置繁琐 |

本文采用 **pnpm + TurboRepo** 方案。

## 二、项目初始化

### 2.1 创建项目结构

```bash
# 1. 创建项目根目录
mkdir micro-frontend-monorepo
cd micro-frontend-monorepo

# 2. 初始化 pnpm
pnpm init

# 3. 安装 TurboRepo
pnpm add -D turbo

# 4. 创建目录结构
mkdir -p apps packages/main packages/shared packages/types packages/config
```

```yaml
# pnpm-workspace.yaml
packages:

  - 'packages/main'

  # 微应用
  - 'apps/*'

  - 'packages/*'
```

```typescript
// packages/shared/src/utils/index.ts
/**
 * 格式化日期
 */
export function formatDate(date: Date, format: string = 'YYYY-MM-DD'): string {
  const year = date.getFullYear()
  const month = String(date.getMonth() + 1).padStart(2, '0')
  const day = String(date.getDate()).padStart(2, '0')

  return format
    .replace('YYYY', String(year))
    .replace('MM', month)
    .replace('DD', day)
}

/**
 * 防抖函数
 */
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timer: ReturnType<typeof setTimeout>
  return function (this: any, ...args: Parameters<T>) {
    clearTimeout(timer)
    timer = setTimeout(() => fn.apply(this, args), delay)
  }
}

/**
 * 节流函数
 */
export function throttle<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let lastTime = 0
  return function (this: any, ...args: Parameters<T>) {
    const now = Date.now()
    if (now - lastTime >= delay) {
      lastTime = now
      fn.apply(this, args)
    }
  }
}

/**
 * 深拷贝
 */
export function deepClone<T>(obj: T): T {
  if (obj === null || typeof obj !== 'object') return obj
  if (obj instanceof Date) return new Date(obj.getTime()) as any
  if (obj instanceof Array) return obj.map(item => deepClone(item)) as any
  if (obj instanceof Object) {
    const clonedObj: any = {}
    for (const key in obj) {
      if (obj.hasOwnProperty(key)) {
        clonedObj[key] = deepClone(obj[key])
      }
    }
    return clonedObj
  }
  return obj
}
```

```typescript
// packages/shared/src/request/index.ts
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import type { GlobalState } from '@micro-frontend/types'

// 创建 axios 实例
export const createRequest = (getState?: () => GlobalState) => {
  const instance: AxiosInstance = axios.create({
    baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
    timeout: 10000
  })

  // 请求拦截器
  instance.interceptors.request.use(
    (config) => {
      // 添加 token
      const state = getState?.()
      if (state?.token) {
        config.headers.Authorization = `Bearer ${state.token}`
      }
      return config
    },
    (error) => Promise.reject(error)
  )

  // 响应拦截器
  instance.interceptors.response.use(
    (response: AxiosResponse) => response.data,
    (error) => {
      // 统一错误处理
      if (error.response?.status === 401) {
        // token 过期，触发登出
        window.dispatchEvent(new CustomEvent('micro-fe:logout'))
      }
      return Promise.reject(error)
    }
  )

  return instance
}

// 默认请求实例
export const request = createRequest()
```

### 4.2 主应用入口

```typescript
// packages/main/src/main.ts
import { createApp } from 'vue'
import { createRouter, createWebHistory } from 'vue-router'
import { registerMicroApps, start, initGlobalState } from 'qiankun'
import App from './App.vue'
import { MICRO_APP_NAMES, APP_PORTS } from '@micro-frontend/shared/constants'
import type { GlobalState, MicroAppConfig } from '@micro-frontend/types'

// 创建路由
const router = createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: '/',
      name: 'Home',
      component: () => import('./views/Home.vue')
    }
  ]
})

const app = createApp(App)
app.use(router)
app.mount('#app')

// 初始化全局状态
const initialState: GlobalState = {
  user: null,
  token: '',
  theme: 'light',
  language: 'zh-CN'
}

const actions = initGlobalState(initialState)

// 主应用监听状态变化
actions.onGlobalStateChange((state, prev) => {
  console.log('[主应用] 状态变化:', state, prev)

  // 同步主题
  document.documentElement.setAttribute('data-theme', state.theme)

  // 同步语言
  document.documentElement.lang = state.language
})

// 注册微应用
const microApps: MicroAppConfig[] = [
  {
    name: MICRO_APP_NAMES.VUE,
    entry: `//localhost:${APP_PORTS.VUE}`,
    container: '#subapp-viewport',
    activeRule: '/vue',
    props: {
      actions,
      routerBase: '/vue'
    }
  },
  {
    name: MICRO_APP_NAMES.REACT,
    entry: `//localhost:${APP_PORTS.REACT}`,
    container: '#subapp-viewport',
    activeRule: '/react',
    props: {
      actions,
      routerBase: '/react'
    }
  },
  {
    name: MICRO_APP_NAMES.ADMIN,
    entry: `//localhost:${APP_PORTS.ADMIN}`,
    container: '#subapp-viewport',
    activeRule: '/admin',
    props: {
      actions,
      routerBase: '/admin'
    }
  }
]

registerMicroApps(microApps, {
  beforeLoad: (app) => {
    console.log(`[qiankun] 加载 ${app.name}`)
    return Promise.resolve()
  },
  beforeMount: (app) => {
    console.log(`[qiankun] 挂载 ${app.name}`)
    return Promise.resolve()
  },
  afterMount: (app) => {
    console.log(`[qiankun] 挂载完成 ${app.name}`)
    return Promise.resolve()
  },
  beforeUnmount: (app) => {
    console.log(`[qiankun] 卸载 ${app.name}`)
    return Promise.resolve()
  },
  afterUnmount: (app) => {
    console.log(`[qiankun] 卸载完成 ${app.name}`)
    return Promise.resolve()
  }
})

// 启动 qiankun
start({
  prefetch: true,
  sandbox: {
    experimentalStyleIsolation: true
  }
})
```

```typescript
import { createApp } from 'vue'
import { createRouter, createWebHistory } from 'vue-router'
import App from './App.vue'
import routes from './router'
import { qiankunWindow, qiankunUnmount } from 'vite-plugin-qiankun/dist/helper'
import type { MicroAppProps, GlobalState } from '@micro-frontend/types'

let app: any
let router: any
let history: any

export async function bootstrap() {
  console.log('[Vue 子应用] bootstrap')
}

export async function mount(props: MicroAppProps) {
  console.log('[Vue 子应用] mount', props)

  const { container, routerBase, onStateChange } = props

  // 创建路由
  history = createWebHistory(
    qiankunWindow.__POWERED_BY_QIANKUN__ ? routerBase : '/'
  )
  router = createRouter({ history, routes })

  // 监听全局状态变化
  onStateChange((state: GlobalState) => {
    console.log('[Vue 子应用] 状态变化:', state)
    app.config.globalProperties.$globalState = state
  }, true)

  // 创建实例
  app = createApp(App)
  app.use(router)

  const containerElement = container
    ? container.querySelector('#app')
    : document.querySelector('#app')

  app.mount(containerElement)
}

export async function unmount() {
  console.log('[Vue 子应用] unmount')
  app?.unmount()
  history = null
  router = null
  app = null
}

if (!qiankunWindow.__POWERED_BY_QIANKUN__) {
  mount({
    onStateChange: () => {},
    emitEvent: () => {},
    routerBase: '/'
  })
}
```
