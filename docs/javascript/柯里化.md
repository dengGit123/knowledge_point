# JavaScript 柯里化 (Currying)

## 简介

**柯里化（Currying）**是一种将接受多个参数的函数转换为一系列接受单个参数的函数的技术。

简单来说：**把 `f(a, b, c)` 转换为 `f(a)(b)(c)`**

## 什么是柯里化

### 基本概念

```javascript
// 普通函数：一次传入所有参数
function add(a, b, c) {
  return a + b + c;
}

add(1, 2, 3);  // 6

// 柯里化后的函数：每次只传一个参数
function curriedAdd(a) {
  return function(b) {
    return function(c) {
      return a + b + c;
    };
  };
}

curriedAdd(1)(2)(3);  // 6
```

### 图解柯里化

```
普通函数调用：
┌─────────────────────────────────────┐
│         add(1, 2, 3)                │
│              │                      │
│              ▼                      │
│         返回 6                       │
└─────────────────────────────────────┘

柯里化函数调用：
┌─────────────────────────────────────────────────────┐
│  curriedAdd(1)(2)(3)                                │
│       │        │  │                                 │
│       ▼        ▼  ▼                                 │
│  ┌──────────────────────────────────────────────┐  │
│  │ curriedAdd(1)  → 返回函数 f1                  │  │
│  │ f1(2)           → 返回函数 f2                 │  │
│  │ f2(3)           → 返回 6                      │  │
│  └──────────────────────────────────────────────┘  │
└─────────────────────────────────────────────────────┘

记忆方式：俄罗斯套娃
┌─────────────────────────────────────────────────────┐
│                                                       │
│            🎁 (最外层：传入1)                          │
│         ┌─────────────────┐                          │
│         │  🎁 (中层：传入2) │                          │
│         │  ┌───────────┐  │                          │
│         │  │ 🎁 (核心) │  │                          │
│         │  │ (传入3)   │  │                          │
│         │  │ 返回 6    │  │                          │
│         │  └───────────┘  │                          │
│         └─────────────────┘                          │
│                                                       │
└─────────────────────────────────────────────────────┘
```

## 手动实现柯里化

### 基础版本

```javascript
// 原始函数
function add(a, b, c) {
  return a + b + c;
}

// 手动柯里化
function curriedAdd(a) {
  return function(b) {
    return function(c) {
      return a + b + c;
    };
  };
}

// 使用
curriedAdd(1)(2)(3);  // 6

// 也可以分步调用
const add1 = curriedAdd(1);
const add1And2 = add1(2);
const result = add1And2(3);  // 6
```

### ES6 箭头函数写法

```javascript
// 更简洁的写法
const curriedAdd = a => b => c => a + b + c;

// 等价于：
// const curriedAdd = function(a) {
//   return function(b) {
//     return function(c) {
//       return a + b + c;
//     };
//   };
// };

curriedAdd(1)(2)(3);  // 6
```

### 通用柯里化函数

```javascript
// 编写一个通用的柯里化函数
function curry(fn) {
  return function curried(...args) {
    // 如果传入的参数数量 >= 原函数需要的参数数量
    if (args.length >= fn.length) {
      // 直接执行原函数
      return fn.apply(this, args);
    } else {
      // 否则返回一个新函数，继续接收参数
      return function(...moreArgs) {
        return curried.apply(this, [...args, ...moreArgs]);
      };
    }
  };
}

// 使用示例
function add(a, b, c) {
  return a + b + c;
}

const curriedAdd = curry(add);

// 可以一次性传入所有参数
curriedAdd(1, 2, 3);  // 6

// 也可以分步传入
curriedAdd(1)(2, 3);  // 6
curriedAdd(1)(2)(3);  // 6

// 可以分步保存中间结果
const add1 = curriedAdd(1);
const add1And2 = add1(2);
add1And2(3);  // 6
```

## 柯里化的理解方式

### 1. 渐进式传递参数

```javascript
// 就像填空题，一步一步填写
const greet = curry((greeting, name, punctuation) => {
  return `${greeting}, ${name}${punctuation}`;
});

// 第一步：确定问候语
const sayHello = greet("Hello");

// 第二步：确定名字
const sayHelloToJohn = sayHello("John");

// 第三步：确定标点
sayHelloToJohn("!");  // "Hello, John!"
sayHelloToJohn("~");  // "Hello, John~"

// 也可以跳过中间步骤
greet("Hi")("Jane")("?");  // "Hi, Jane?"
```

### 2. 函数工厂

```javascript
// 柯里化就像工厂，可以批量生产"专用函数"
const multiply = (a, b) => a * b;
const curriedMultiply = curry(multiply);

// 生产专用函数
const double = curriedMultiply(2);   // 乘以2的函数
const triple = curriedMultiply(3);   // 乘以3的函数
const quadruple = curriedMultiply(4); // 乘以4的函数

// 使用
double(5);      // 10
triple(5);      // 15
quadruple(5);   // 20
```

### 3. 参数复用

```javascript
// 场景：创建不同折扣的计算函数
const calculateDiscount = curry((discount, price) => {
  return price * (1 - discount);
});

// 创建不同折扣的专用函数
const with10PercentOff = calculateDiscount(0.1);
const with20PercentOff = calculateDiscount(0.2);
const with30PercentOff = calculateDiscount(0.3);

// 使用
with10PercentOff(100);  // 90
with20PercentOff(100);  // 80
with30PercentOff(100);  // 70
```

## 柯里化的实际应用

### 1. 创建可复用的函数

```javascript
// 数据过滤
const filter = curry((predicate, array) => array.filter(predicate));

// 创建专用过滤函数
const filterOdds = filter(x => x % 2 !== 0);    // 过滤奇数
const filterEvens = filter(x => x % 2 === 0);    // 过滤偶数
const filterPositive = filter(x => x > 0);      // 过滤正数

const numbers = [-2, -1, 0, 1, 2, 3, 4, 5];

filterOdds(numbers);     // [-1, 1, 3, 5]
filterEvens(numbers);    // [-2, 0, 2, 4]
filterPositive(numbers); // [1, 2, 3, 4, 5]
```

### 2. 数据映射

```javascript
const map = curry((fn, array) => array.map(fn));

// 创建专用映射函数
const mapToUpperCase = map(s => s.toUpperCase());
const mapToDouble = map(x => x * 2);
const mapToString = map(x => String(x));

mapToUpperCase(['a', 'b', 'c']);  // ['A', 'B', 'C']
mapToDouble([1, 2, 3]);           // [2, 4, 6]
mapToString([1, 2, 3]);           // ['1', '2', '3']
```

### 3. DOM 操作

```javascript
// 创建元素选择器
const querySelector = curry((selector, element) => {
  return element.querySelector(selector);
});

// 绑定 document
const query = querySelector(document);
const getMain = query('main');
const getHeader = query('header');

// 绑定其他元素
const container = document.querySelector('.container');
const queryContainer = querySelector(container);
const getItem = queryContainer('.item');
```

### 4. 表单验证

```javascript
// 验证函数
const validate = curry((rules, value) => {
  return rules.every(rule => rule(value));
});

// 定义规则
const isRequired = value => value && value.length > 0;
const isEmail = value => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
const minLength = min => value => value && value.length >= min;
const maxLength = max => value => value && value.length <= max;

// 创建验证器
const validateUsername = validate([
  isRequired,
  minLength(3),
  maxLength(16)
]);

const validateEmail = validate([
  isRequired,
  isEmail
]);

validateUsername('john');       // true
validateUsername('jo');         // false (太短)
validateEmail('test@test.com'); // true
validateEmail('invalid');       // false
```

### 5. API 请求

```javascript
// 创建 API 请求函数
const request = curry((method, url, data) => {
  return fetch(url, {
    method,
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  }).then(res => res.json());
});

// 创建专用请求方法
const get = request('GET');
const post = request('POST');
const put = request('PUT');
const del = request('DELETE');

// 创建 API 端点
const getUser = get('/api/user');
const createUser = post('/api/user');
const updateUser = put('/api/user');
const deleteUser = del('/api/user');

// 使用
getUser();  // GET /api/user
createUser({ name: 'John' });  // POST /api/user with data
```

### 6. 函数组合

```javascript
// 柯里化使函数组合更容易
const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x);

const toUpperCase = s => s.toUpperCase();
const exclaim = s => s + '!';
const repeat = s => s + s;

// 组合函数
const shout = compose(exclaim, toUpperCase);
const excitedShout = compose(repeat, shout);

shout('hello');       // 'HELLO!'
excitedShout('hello'); // 'HELLO!HELLO!'
```

### 7. 日志记录

```javascript
const log = curry((level, message, data) => {
  const timestamp = new Date().toISOString();
  console[level](`[${timestamp}] ${message}`, data);
});

// 创建专用日志函数
const logInfo = log('info');
const logWarn = log('warn');
const logError = log('error');

// 使用
logInfo('用户登录', { userId: 123 });
logWarn('内存使用率高', { usage: '90%' });
logError('请求失败', { error: 'Connection timeout' });
```

### 8. 数组操作链

```javascript
const filter = curry((fn, arr) => arr.filter(fn));
const map = curry((fn, arr) => arr.map(fn));
const reduce = curry((fn, init, arr) => arr.reduce(fn, init));

// 数据处理管道
const processNumbers = compose(
  reduce((sum, n) => sum + n, 0),
  map(n => n * 2),
  filter(n => n > 2)
);

processNumbers([1, 2, 3, 4, 5]);  // (3,4,5) -> (6,8,10) -> 24
```

## 高级柯里化技巧

### 占位符支持

```javascript
// 实现 lodash 风格的占位符
const _ = {};

function curry(fn, placeholder = _) {
  return function curried(...args) {
    // 替换占位符
    const replacePlaceholder = (args, newArgs) => {
      const result = [...args];
      let placeholderIndex = result.indexOf(placeholder);
      let newIndex = 0;

      while (placeholderIndex !== -1 && newIndex < newArgs.length) {
        result[placeholderIndex] = newArgs[newIndex];
        placeholderIndex = result.indexOf(placeholder, placeholderIndex + 1);
        newIndex++;
      }

      return [...result, ...newArgs.slice(newIndex)];
    };

    // 检查是否有占位符
    const hasPlaceholder = args.some(arg => arg === placeholder);
    const validArgs = hasPlaceholder ? args.filter(arg => arg !== placeholder) : args;
    const hasEnough = validArgs.length >= fn.length;

    if (hasEnough && !hasPlaceholder) {
      return fn.apply(this, args);
    }

    return function(...newArgs) {
      const combined = replacePlaceholder(args, newArgs);
      return curried.apply(this, combined);
    };
  };
}

// 使用示例
const subtract = (a, b, c) => a - b - c;
const curriedSubtract = curry(subtract);

// 占位符跳过第一个参数，先指定后面的参数
const subtractFrom10 = curriedSubtract(_, 10);
// 然后传入第一个参数
subtractFrom10(20);  // 20 - 10 = 0 (如果c未定义)
```

### 反柯里化（Uncurrying）

```javascript
// 反柯里化：将柯里化函数转回普通函数
function uncurry(fn) {
  return function(...args) {
    return args.reduce((f, arg) => f(arg), fn);
  };
}

const curriedAdd = a => b => c => a + b + c;
const add = uncurry(curriedAdd);

add(1, 2, 3);  // 6
add(1, 2);     // c => 1 + 2 + c
```

### 部分应用 vs 柯里化

```javascript
// 部分应用（Partial Application）
// 固定部分参数，剩余参数一次性传入
const partial = (fn, ...presetArgs) => (...laterArgs) => {
  return fn(...presetArgs, ...laterArgs);
};

function add(a, b, c) {
  return a + b + c;
}

const add1And2 = partial(add, 1, 2);
add1And2(3);  // 6

// 区别：
// 柯里化：每次只能传一个参数，直到传完
// 部分应用：可以一次固定多个参数
```

## 柯里化的作用

### 1. 参数复用

```javascript
// 避免重复传递相同的参数
const fetch = curry((baseUrl, endpoint, params) => {
  return fetch(`${baseUrl}${endpoint}?${params}`);
});

const apiFetch = fetch('https://api.example.com');
const getUsers = apiFetch('/users');
const getUserById = userId => getUsers(`id=${userId}`);

getUserById(123);  // 复用了 baseUrl
```

### 2. 延迟执行

```javascript
// 先配置，后执行
const calculate = curry((discount, tax, price) => {
  return price * (1 - discount) * (1 + tax);
});

// 配置
const summerSale = calculate(0.2);  // 20% 折扣
const withTax = summerSale(0.08);   // 8% 税

// 后续随时使用
withTax(100);  // 86.4
withTax(200);  // 172.8
```

### 3. 代码可读性

```javascript
// 普通写法
const result = map(filter(numbers, n => n > 0), n => n * 2);

// 柯里化写法（更清晰）
const result = pipe(
  filter(n => n > 0),
  map(n => n * 2)
)(numbers);
```

### 4. 函数组合

```javascript
// 柯里化使函数更容易组合
const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);

const process = pipe(
  filter(n => n % 2 === 0),
  map(n => n * 2),
  reduce((a, b) => a + b, 0)
);

process([1, 2, 3, 4, 5, 6]);  // 2*2 + 4*2 + 6*2 = 24
```

## 柯里化的优缺点

### 优点

| 优点 | 说明 |
|------|------|
| **参数复用** | 避免重复传递相同参数 |
| **延迟执行** | 先配置后执行，更灵活 |
| **函数组合** | 更容易组合复杂操作 |
| **代码复用** | 创建可复用的专用函数 |
| **可读性** | 声明式代码风格 |

### 缺点

| 缺点 | 说明 |
|------|------|
| **性能开销** | 多层函数调用，嵌套越深开销越大 |
| **调试困难** | 调用栈变深，追踪问题变难 |
| **学习曲线** | 需要理解函数式编程概念 |
| **JS 不原生支持** | 需要自己实现或使用库 |
| **部分场景不适用** | 简单操作反而增加复杂度 |

## 实际项目示例

### Vue 中的柯里化应用

```javascript
// 创建表单验证器
export function createValidator(rules) {
  return curry((fieldName, value) => {
    const fieldRules = rules[fieldName] || [];
    const errors = [];

    for (const rule of fieldRules) {
      if (!rule.validator(value)) {
        errors.push(rule.message);
      }
    }

    return errors.length > 0 ? errors : null;
  });
}

// 使用
const validate = createValidator({
  username: [
    { validator: v => v && v.length >= 3, message: '用户名至少3个字符' },
    { validator: v => v && v.length <= 16, message: '用户名最多16个字符' }
  ],
  email: [
    { validator: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v), message: '邮箱格式不正确' }
  ]
});

const validateUsername = validate('username');
const validateEmail = validate('email');

validateUsername('john');  // null（通过）
validateUsername('jo');    // ['用户名至少3个字符']
validateEmail('test@test.com');  // null（通过）
```

### React 中的柯里化应用

```javascript
// 事件处理器柯里化
import { curry } from 'lodash';

const handleChange = curry((fieldName, setValue, event) => {
  setValue(event.target.value);
});

function MyForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');

  return (
    <form>
      <input
        onChange={handleChange('name', setName)}
        value={name}

        onChange={handleChange('email', setEmail)}

    </form>
  );
}
```

## 常见库的实现

### Lodash 的 curry

```javascript
import _ from 'lodash';

// Lodash 的 curry 功能更强大
const abc = function(a, b, c) {
  return [a, b, c];
};

const curried = _.curry(abc);

curried(1)(2)(3);  // [1, 2, 3]
curried(1, 2)(3);  // [1, 2, 3]
curried(1)(2, 3);  // [1, 2, 3]
curried(1, 2, 3);  // [1, 2, 3]

// 占位符支持
const curriedWithPlaceholder = _.curry(abc);
const c = curriedWithPlaceholder(1, _, 3);
c(2);  // [1, 2, 3]
```

### Ramda 的自动柯里化

```javascript
import R from 'ramda';

// Ramda 的函数默认都是柯里化的
const add = R.add;
const multiply = R.multiply;

const add5 = add(5);
const double = multiply(2);

add5(10);    // 15
double(5);   // 10

// 组合
const addThenDouble = R.pipe(add5, double);
addThenDouble(10);  // 30
```

## 总结

### 柯里化记忆口诀

```
多参函数变单参，层层嵌套返回新
参数复用效率高，函数组合更灵活
配置执行分两步，代码清晰易维护
性能开销需考虑，适当使用是关键
```

### 使用建议

```javascript
// ✅ 适合使用柯里化的场景
1. 需要创建多个相似功能的函数
2. 需要复用部分参数
3. 进行函数组合
4. 需要延迟执行（先配置后使用）

// ❌ 不适合使用柯里化的场景
1. 一次性调用的简单函数
2. 参数之间有强依赖关系
3. 对性能要求极高的场景
4. 团队成员不熟悉函数式编程
```
