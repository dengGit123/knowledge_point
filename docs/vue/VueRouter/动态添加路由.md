### 一、概述

Vue Router 支持在应用运行时**动态添加或移除路由**。最常见的场景是**权限控制**——根据用户角色动态注册有权限访问的路由，没有权限的路由根本不存在于路由表中。

> 📖 [Vue Router - 动态路由](https://router.vuejs.org/zh/guide/advanced/dynamic-routing.html)

### 二、添加路由（addRoute）

#### 1. 添加顶级路由

```js
const router = createRouter({ /* ... */ })

router.addRoute({
  path: '/admin',
  name: 'admin',
  component: () => import('@/views/AdminView.vue'),
})
```

#### 2. 添加嵌套子路由

第一个参数传**父路由的 name**，第二个参数传子路由配置：

```js
// 父路由已定义：{ path: '/dashboard', name: 'dashboard', component: ... }

router.addRoute('dashboard', {
  path: 'analytics',
  name: 'dashboard-analytics',
  component: () => import('@/views/AnalyticsView.vue'),
})
// 效果等同 /dashboard/analytics
```

### 三、移除路由（removeRoute）

通过路由的 **name** 移除：

```js
router.removeRoute('admin')
```

### 四、查询路由

```js
// 检查路由是否存在
router.hasRoute('admin') // true / false

// 获取所有路由记录
router.getRoutes() // [{ path: '/', name: 'home', ... }, ...]
```

### 五、实际应用：权限路由

根据用户角色动态注册路由：

```js
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'

// 公共路由（所有人都能访问）
const publicRoutes = [
  { path: '/', name: 'home', component: () => import('@/views/HomeView.vue') },
  { path: '/login', name: 'login', component: () => import('@/views/LoginView.vue') },
  { path: '/:pathMatch(.*)*', name: 'not-found', component: () => import('@/views/NotFoundView.vue') },
]

// 管理员路由
const adminRoutes = [
  {
    path: '/admin',
    name: 'admin',
    component: () => import('@/views/AdminLayout.vue'),
    children: [
      { path: '', name: 'admin-home', component: () => import('@/views/AdminHome.vue') },
      { path: 'users', name: 'admin-users', component: () => import('@/views/AdminUsers.vue') },
    ],
  },
]

// 编辑者路由
const editorRoutes = [
  {
    path: '/editor',
    name: 'editor',
    component: () => import('@/views/EditorLayout.vue'),
    children: [
      { path: '', name: 'editor-home', component: () => import('@/views/EditorHome.vue') },
      { path: 'posts', name: 'editor-posts', component: () => import('@/views/EditorPosts.vue') },
    ],
  },
]

const router = createRouter({
  history: createWebHistory(),
  routes: publicRoutes,
})

// 根据角色动态添加路由
export function setupDynamicRoutes(role) {
  // 先清除之前可能添加的路由
  removeDynamicRoutes()

  if (role === 'admin') {
    adminRoutes.forEach((route) => router.addRoute(route))
  }

  if (role === 'editor' || role === 'admin') {
    editorRoutes.forEach((route) => router.addRoute(route))
  }
}

function removeDynamicRoutes() {
  // 移除管理员路由
  adminRoutes.forEach((route) => {
    if (route.name && router.hasRoute(route.name)) {
      router.removeRoute(route.name)
    }
    // 移除子路由
    route.children?.forEach((child) => {
      if (child.name && router.hasRoute(child.name)) {
        router.removeRoute(child.name)
      }
    })
    if (route.name && router.hasRoute(route.name)) {
      router.removeRoute(route.name)
    }
  })

  // 移除编辑者路由
  editorRoutes.forEach((route) => {
    route.children?.forEach((child) => {
      if (child.name && router.hasRoute(child.name)) {
        router.removeRoute(child.name)
      }
    })
    if (route.name && router.hasRoute(route.name)) {
      router.removeRoute(route.name)
    }
  })
}

export default router
```

```js
// 登录成功后调用
import { setupDynamicRoutes } from '@/router'

async function handleLogin() {
  const { token, role } = await login(credentials)
  localStorage.setItem('token', token)

  // 根据角色动态注册路由
  setupDynamicRoutes(role)

  // 跳转到首页或之前想去的页面
  router.push(redirect || '/')
}
```

### 六、注意事项

```js
// ⚠️ addRoute 是即时生效的，但如果用户当前正好在被添加的路由上，
// 需要用 router.replace 或 router.push 触发导航

// ⚠️ 添加路由时要确保 name 唯一，重复 name 会覆盖旧路由

// ⚠️ 404 路由（/:pathMatch(.*)*）要放在初始路由表的最后
// 动态添加的路由优先级高于 404，所以不必担心顺序问题
```
